import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class Ball {
    // execute application
    public static void main(String args[]) {
        JFrame frame = new JFrame("Parabolic Motion");
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        BallPanel bp = new BallPanel();
        frame.add(bp);
        frame.setSize(800, 600); // set frame size
        frame.setVisible(true); // display frame
    } // end main
}

// class BallPanel
class BallPanel extends JPanel implements ActionListener {
    private int delay = 10;
    protected Timer timer;

    private double x = 20;        // x position
    private double y = 0;        // y position
    private int radius = 15;     // ball radius

    private double vx = 5;     // x velocity 
    private double vy = 0;     // y velocity 
    private double gravity = 0.1; 

    public BallPanel() {
        timer = new Timer(delay, this);
        timer.start();        // start the timer
    }

    public void actionPerformed(ActionEvent e) {
        move();
        repaint();
    }

    private void move() {
        x += vx;
        vy += gravity; // Increment vertical velocity by gravity
        y += vy;
        
    }

    // draw rectangles and arcs
    public void paintComponent(Graphics g) {
        super.paintComponent(g); // call superclass's paintComponent 
        g.setColor(Color.red);

        // adjust ball position
        g.fillOval((int)(x - radius), (int)(y - radius), radius * 2, radius * 2);
    }
}
